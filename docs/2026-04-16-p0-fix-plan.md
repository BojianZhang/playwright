## 修复 3：代理互斥锁（Proxy Mutex）

### 抽象层级

互斥锁属于**调度能力**，应放在 `shared-batch-orchestration`：

```
shared-batch-orchestration/
├── batch-orchestration.js   ← 并发主循环（已有）
├── task-queue.js            ← 任务队列（已有）
├── worker-state.js          ← Worker 状态（已有）
└── mutex.js                 ← 【新增】通用互斥锁工具
```

### mutex.js 设计

```js
// shared-batch-orchestration/mutex.js
// 边界：只提供内存级互斥原语，不做任何业务判断

/**
 * 创建一个串行互斥锁（Promise 链式）
 * 同一时刻只允许一个任务执行
 */
function createMutex() {
  let current = Promise.resolve();
  return async (task) => {
    const run = current.then(() => task());
    current = run.catch(() => {});
    return run;
  };
}

/**
 * 创建代理独占池
 * 同一条代理同一时刻只允许一个 Worker 占用
 */
function createProxyLockSet() {
  const locked = new Set();
  return {
    tryAcquire(proxyKey) {
      if (locked.has(proxyKey)) return false;
      locked.add(proxyKey);
      return true;
    },
    release(proxyKey) {
      locked.delete(proxyKey);
    },
    isLocked(proxyKey) {
      return locked.has(proxyKey);
    },
  };
}

module.exports = { createMutex, createProxyLockSet };
```

### batch-runner 接入方式

```js
const { createProxyLockSet } = require('../../shared-batch-orchestration/mutex');

// createBatchRunContext 里初始化
batchContext.proxies.lockSet = createProxyLockSet();

// acquireNextProxy 里加锁
function acquireNextProxy(batchContext) {
  const list = batchContext?.proxies?.list || [];
  const lockSet = batchContext?.proxies?.lockSet;
  const records = batchContext?.proxies?.healthStore?.records || {};

  // 过滤：硬封禁 + 当前被占用的
  const activeList = list.filter(proxy => {
    const key = proxy?.proxyKey || '';
    const record = records[key] || {};
    return !isProxyHardBlocked(record) && !lockSet?.isLocked(key);
  });

  // 按 tier 排序后取第一个可用的
  const sorted = activeList.sort((a, b) => getProxySelectionTier(b) - getProxySelectionTier(a));
  const proxy = sorted[0] || null;
  if (proxy && lockSet) lockSet.tryAcquire(proxy.proxyKey);
  return proxy;
}

// processBatchTask finally 里释放
finally {
  if (proxy?.proxyKey && batchContext.proxies.lockSet) {
    batchContext.proxies.lockSet.release(proxy.proxyKey);
  }
}
```

## 修复 4：运行时代理热剔除

### 触发条件（从失败结果判断）

```js
// Dreamina/0.0.3/failure-classifier.js（新建）

// 代理级别的硬失败 → 立即从内存池移除
const PROXY_HARD_FAILURE_REASONS = [
  'DREAMINA_PROXY_CONNECTIVITY_FAILED',
  'DREAMINA_PROXY_TARGET_UNREACHABLE',
  'DREAMINA_PROXY_TLS_FAILED',
  'DREAMINA_ENTRY_WHITE_SCREEN',
  'DREAMINA_ENTRY_DEAD_PAGE',
  'SITE_ENTRY_OPEN_RETRY_EXHAUSTED',
];

// 业务级别的失败 → 不惩罚代理
const BUSINESS_FAILURE_REASONS = [
  'ACCOUNT_ALREADY_EXISTS',
  'ACCOUNT_ALREADY_EXISTS_PRECHECK',
  'DREAMINA_CREDENTIAL_NO_STATE_CHANGE',
  'DREAMINA_VERIFICATION_WRONG_CODE',
  'DREAMINA_VERIFICATION_CODE_RATE_LIMITED',
];

function isProxyHardFailure(reason) {
  return PROXY_HARD_FAILURE_REASONS.includes(String(reason || '').toUpperCase());
}

function isBusinessFailure(reason) {
  return BUSINESS_FAILURE_REASONS.some(r => String(reason || '').toUpperCase().includes(r));
}
```

### 热剔除逻辑接入点

```js
// processBatchTask 失败处理里
if (isProxyHardFailure(result?.finalReason) && config.proxy?.evictOnHardFailure) {
  evictProxyFromPool(batchContext, proxy.proxyKey);
  console.log(`[Proxy Evict] ${proxy.proxyKey} 已从运行时池移除 | reason=${result.finalReason}`);
}

// 热剔除函数（加在 batch-runner.js 里）
function evictProxyFromPool(batchContext, proxyKey) {
  batchContext.proxies.list = batchContext.proxies.list.filter(
    p => p?.proxyKey !== proxyKey
  );
  // 同时释放可能的锁
  batchContext.proxies.lockSet?.release(proxyKey);
}
```
